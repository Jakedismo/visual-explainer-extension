"Design a system that scales horizontally while maintaining strong consistency for user sessions."
session-manager.ts
export class SessionManager {
  constructor(private redis: Redis) {}

  async create(userId: string) {
    const token = crypto.randomUUID();
    await this.redis.set(`session:${token}`, userId, 'EX', 3600);
    return token;
  }
}
